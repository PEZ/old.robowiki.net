package cx.femto;

import robocode.AdvancedRobot;
import robocode.ScannedRobotEvent;
/**
 * @author iiley
 * codesize 42
 */
public class Yaya extends AdvancedRobot
{
    public void run()
    {
		do{
           turnRight(10);
	       setAhead(Math.sin(getTime()/8)*1200d);
		}while(true);
    }

    public void onScannedRobot(ScannedRobotEvent e)
    {
        setFire(3d);
    }
}
