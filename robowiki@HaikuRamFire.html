package tango.haiku;
import robocode.*;

public class HaikuRamFire extends AdvancedRobot
{
	String TargetName;
	public void run() {
		setTurnGunRight(Double.POSITIVE_INFINITY);
		setAhead(Double.POSITIVE_INFINITY);		
	}


	public void onScannedRobot(ScannedRobotEvent e) {
			fire(3);
			turnRight(e.getBearing()); 
	}

}
