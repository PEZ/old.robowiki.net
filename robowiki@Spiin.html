package cx.femto;

import robocode.AdvancedRobot;
import robocode.ScannedRobotEvent;
/**
 * @author iiley
 * codesize 29
 */
public class Spiin extends AdvancedRobot {
    public void run() {
        do {
            setAhead(12d);
            turnRight(10d);
        } while(true);
    }

    public void onScannedRobot(ScannedRobotEvent e) {
        setFire(3d);
    }
}
