	public double getBearingTo(String part , ScannedRobotEvent e) {
		if (part.equals("Gun"))
			return(e.getBearing() - (getGunHeading() - getHeading()));
		else
			return(e.getBearing() - (getRadarHeading() - getHeading()));
	}
