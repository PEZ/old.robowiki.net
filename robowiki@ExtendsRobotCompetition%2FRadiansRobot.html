package wiki;

import robocode.Robot;

public class RadiansRobot extends Robot {

	public void turnRadarLeftRadians(double angle) {
		turnRadarLeft(Math.toDegrees(angle));
	}
	
	public void turnRadarRightRadians(double angle) {
		turnRadarRight(Math.toDegrees(angle));
	}
	
	public void turnGunLeftRadians(double angle) {
		turnGunLeft(Math.toDegrees(angle));
	}

	public void turnGunRightRadians(double angle) {
		turnGunRight(Math.toDegrees(angle));
	}
	
	public void turnLeftRadians(double angle) {
		turnLeft(Math.toDegrees(angle));
	}

	public void turnRightRadians(double angle) {
		turnRight(Math.toDegrees(angle));
	}
	
	public double getRadarHeadingRadians(){
		return Math.toRadians(getRadarHeading());
	}
	
	public double getGunHeadingRadians(){
		return Math.toRadians(getGunHeading());
	}
	
	public double getHeadingRadians(){
		return Math.toRadians(getHeading());
	}
}
